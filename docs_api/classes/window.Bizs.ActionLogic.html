function ajaxDelCallback( _d, _ins ){
   var _trigger = $(this);
   if( _d && !_d.errorno ){
       JC.msgbox( _d.errmsg || '操作成功', _trigger, 0, function(){
           JC.f.reloadPage( '?usercallback=ajaxaction' );
       });
   }else{
       JC.Dialog.alert( _d && _d.errmsg ? _d.errmsg : '操作失败, 请重试!' , 1 );
   }
}
